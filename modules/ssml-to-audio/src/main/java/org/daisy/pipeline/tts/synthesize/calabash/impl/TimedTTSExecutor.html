20  class TimedTTSExecutor {
21  
22  	private static final int FIRST_CHARACTERS = 2500;
23  	private Map<TTSEngine,Integer> totalCharacters = new HashMap<>(); // only count first 2500 characters
24  	private Map<TTSEngine,Integer> maxMicrosecPerCharacter = new HashMap<>();
25  
26  	/**
27  	 * The maximum number of milliseconds the TTS engine is allowed to spend on a single word. This
28  	 * takes into account {@see TTSEngine#expectedMillisecPerWord()}, the number of words processed
29  	 * so far, and the maximum number of milliseconds the engine has spent on a word so far. A large
30  	 * enough safety factor is taken into account for long words and other deviations.
31  	 */
32  	private int maximumMillisec(TTSEngine engine, int sentenceSize) {
33  		int wordCount = 1 + sentenceSize / 6; // ~6 characters/word
34  		Integer n = totalCharacters.get(engine);
35  		if (n == null)
36  			// initially base timeout on engine.expectedMillisecPerWord()
37  			return 10 * engine.expectedMillisecPerWord() * wordCount;
38  		else {
39  			// in the long run base timeout on maxActualMicrosecPerCharacter
40  			Integer maxActualMicrosecPerCharacter = maxMicrosecPerCharacter.get(engine);
41  			if (n < FIRST_CHARACTERS)
42  				// interpolate
43  				return 1
44  					+ 3 * n * maxActualMicrosecPerCharacter * sentenceSize / 1000 / FIRST_CHARACTERS
45  					+ 10 * (FIRST_CHARACTERS - n) * engine.expectedMillisecPerWord() * wordCount / FIRST_CHARACTERS;
46  			else
47  				return 1 + 3 * maxActualMicrosecPerCharacter * sentenceSize / 1000;
48  		}
49  	}
50  
51  	public Collection<AudioBuffer> synthesizeWithTimeout(
52  		TTSTimeout timeout, TTSTimeout.ThreadFreeInterrupter interrupter, TTSLog.Entry log,
53  		String sentence, XdmNode xmlSentence, int sentenceSize, TTSEngine engine, Voice voice,
54  		TTSResource threadResources, List<Mark> marks, List<String> expectedMarks,
55  		AudioBufferAllocator bufferAllocator, boolean retry
56  	) throws SynthesisException, MemoryException, TimeoutException {
57  		long startTime = System.currentTimeMillis();
58  		int timeoutSec = maximumMillisec(engine, sentenceSize) / 1000;
59  		if (log != null)
60  			log.setTimeout(timeoutSec);
61  		try {
62  			timeout.enableForCurrentThread(interrupter, timeoutSec);
63  			Collection<AudioBuffer> result = engine.synthesize(
64  				sentence, xmlSentence, voice, threadResources, marks, expectedMarks, bufferAllocator, retry);
65  			long millisecElapsed = System.currentTimeMillis() - startTime;
66  			synchronized(maxMicrosecPerCharacter) {
67  				Long m = millisecElapsed * 1000 / sentenceSize;
68  				int n = 1000; // don't go below 1 ms
69  				if (maxMicrosecPerCharacter.containsKey(engine))
70  					n = maxMicrosecPerCharacter.get(engine);
71  				else
72  					maxMicrosecPerCharacter.put(engine, n);
73  				if (m > n)
74  					maxMicrosecPerCharacter.put(engine, m.intValue());
75  			}
76  			synchronized(totalCharacters) {
77  				Integer n = totalCharacters.get(engine);
78  				if (n == null) n = 0;
79  				if (n < FIRST_CHARACTERS)
80  					totalCharacters.put(engine, n + sentenceSize);
81  			}
82  			return result;
83  		} catch (InterruptedException e) {
84  			// assuming that the thread was interrupted by the timeout
85  			throw new TimeoutException(timeoutSec, e);
86  		} finally {
87  			timeout.disable();
88  		}
89  	}
90  
91  	static class TimeoutException extends RuntimeException {
92  
93  		private final int seconds;
94  
95  		private TimeoutException(int seconds, InterruptedException cause) {
96  			super("TTS timed out after " + seconds + " seconds", cause);
97  			this.seconds = seconds;
98  		}
99  
100 		/**
101 		 * @return The timeout in seconds.
102 		 */
103 		public int getSeconds() {
104 			return seconds;
105 		}
106 	}
107 }
