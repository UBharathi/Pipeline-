1   package org.daisy.pipeline.tts.synthesize.calabash.impl;
2   
3   import java.util.Collection;
4   import java.util.List;
5   import java.util.Map;
6   import java.util.Set;
7   
8   import net.sf.saxon.s9api.XdmNode;
9   
10  import org.daisy.pipeline.tts.Voice;
11  
12  /**
13   * A class to log structured messages in a big list. TTSLog is meant to be
14   * allocated for every new Pipeline job and written to an external file after
15   * the TTS work of the job is done.
16   */
17  public interface TTSLog {
18  	enum ErrorCode {
19  		UNEXPECTED_VOICE,
20  		AUDIO_MISSING,
21  		CRITICAL_ERROR,
22  		ERROR,
23  		WARNING
24  	}
25  
26  	public static class Error {
27  		public Error(ErrorCode key, String message) {
28  			this.key = key;
29  			this.message = message;
30  		}
31  
32  		public ErrorCode getErrorCode() {
33  			return key;
34  		}
35  
36  		public String getMessage() {
37  			return message;
38  		}
39  
40  		private ErrorCode key;
41  		private String message;
42  	}
43  
44  	interface Entry {
45  		void addError(Error err);
46  
47  		Collection<Error> getReadOnlyErrors();
48  
49  		/**
50  		 * @param ssml is the SSML before being converted to 'ttsinput'
51  		 */
52  		void setSSML(XdmNode ssml);
53  
54  		XdmNode getSSML();
55  
56  		/**
57  		 * @param v is the voice selected by the top-level VoiceManager
58  		 */
59  		void setSelectedVoice(Voice v);
60  
61  		Voice getSelectedVoice();
62  
63  		/**
64  		 * @param v is the actual voice used by the TTS processor (the same as
65  		 *            selectedVoice in the general case, but can be different if
66  		 *            something went wrong)
67  		 */
68  		void setActualVoice(Voice v);
69  
70  		Voice getActualVoice();
71  
72  		/**
73  		 * @param input is the actual string provided as input to the TTS
74  		 *            processor
75  		 */
76  		void addTTSinput(String input);
77  
78  		List<String> getTTSinput();
79  
80  		void resetTTSinput();
81  
82  		/**
83  		 * @param soundfile is a path of a wave, mp3 or ogg file
84  		 */
85  		void setSoundfile(String soundfile);
86  
87  		String getSoundFile();
88  
89  		/**
90  		 * @param secs is the timeout value used while synthesizing the entry
91  		 */
92  		void setTimeout(float secs);
93  
94  		float getTimeout();
95  
96  		/**
97  		 * @param begin offset in seconds
98  		 * @param end offset in seconds
99  		 */
100 		void setPositionInFile(double begin, double end);
101 
102 		double getBeginInFile();
103 
104 		double getEndInFile();
105 	}
106 
107 	/**
108 	 * Supposed to be called within a single-threaded context
109 	 */
110 	Entry getOrCreateEntry(String id);
111 
112 	/**
113 	 * Can be called within a multi-threaded context once all the calls to
114 	 * getOrCreateEntry() are done.
115 	 */
116 	Entry getWritableEntry(String id);
117 
118 	Set<Map.Entry<String, Entry>> getEntries();
119 
120 	void addGeneralError(ErrorCode errcode, String message);
121 
122 	Collection<Error> readonlyGeneralErrors();
123 }
