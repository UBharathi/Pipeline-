1   package org.daisy.common.xproc.calabash.steps;
2   
3   import net.sf.saxon.s9api.QName;
4   import net.sf.saxon.s9api.SaxonApiException;
5   import org.daisy.common.xproc.calabash.XProcStepProvider;
6   
7   import com.xmlcalabash.core.XProcRuntime;
8   import com.xmlcalabash.core.XProcStep;
9   import com.xmlcalabash.runtime.XAtomicStep;
10  
11  public class Message implements XProcStepProvider {
12  
13  	public XProcStep newStep(XProcRuntime runtime, XAtomicStep step) {
14  		return new MessageStep(runtime, step);
15  	}
16  
17  	public static class MessageStep extends com.xmlcalabash.library.Identity {
18  
19  		private static final QName _message = new QName("message");
20  		private static final QName _severity = new QName("severity");
21  		
22  		private MessageStep(XProcRuntime runtime, XAtomicStep step) {
23  			super(runtime, step);
24  		}
25  
26  		@Override
27  		public void run() throws SaxonApiException {
28  			String severity = getOption(_severity, "INFO");
29  			String message = getOption(_message, (String)null);
30  			if ("WARN".equals(severity)) {
31  				warning(this.step.getNode(), message);
32  			} else if ("DEBUG".equals(severity)) {
33  				// DefaultStep#fine() removed in ndw/xmlcalabash1@ce9b07d
34  				this.runtime.getMessageListener().fine(this,this.step.getNode(), message);
35  			} else {
36  				info(this.step.getNode(), message);
37  			}
38  			
39  			super.run();
40  		}
41  	}
42  }
