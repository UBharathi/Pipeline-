1   package org.daisy.pipeline.nlp.breakdetect.calabash.impl;
2   
3   import java.util.Collection;
4   import java.util.Collections;
5   import java.util.HashSet;
6   import java.util.Set;
7   
8   import net.sf.saxon.s9api.QName;
9   
10  /**
11   * Encapsulate what the Lexer needs to know about the input format so it can
12   * parse and rebuild the input documents.
13   */
14  public class FormatSpecifications {
15  
16  	public QName sentenceTag;
17  	public QName wordTag;
18  	public String tmpNsPrefix = "tmp";
19  	public QName langAttr;
20  	public Set<String> inlineElements;
21  	public Set<String> ensureWordBefore;
22  	public Set<String> ensureWordAfter;
23  	public Set<String> ensureSentenceBefore;
24  	public Set<String> ensureSentenceAfter;
25  	public String tmpNs;
26  
27  	FormatSpecifications(String tmpNamespace, String sentenceElement, String wordElement,
28  	        String langNamespace, String langAttr, Collection<String> inlineElements,
29  	        Collection<String> ensureWordBefore, Collection<String> ensureWordAfter,
30  	        Collection<String> ensureSentenceBefore, Collection<String> ensureSentenceAfter) {
31  
32  		if (ensureWordBefore == null)
33  			ensureWordBefore = Collections.EMPTY_LIST;
34  		if (ensureWordAfter == null)
35  			ensureWordAfter = Collections.EMPTY_LIST;
36  		if (ensureSentenceBefore == null)
37  			ensureSentenceBefore = Collections.EMPTY_LIST;
38  		if (ensureSentenceAfter == null)
39  			ensureSentenceAfter = Collections.EMPTY_LIST;
40  
41  		sentenceTag = new QName(tmpNamespace, sentenceElement);
42  		wordTag = new QName(tmpNamespace, wordElement);
43  		this.langAttr = new QName(langNamespace, langAttr);
44  
45  		this.inlineElements = new HashSet<String>(inlineElements);
46  		this.inlineElements.addAll(ensureWordBefore);
47  		this.inlineElements.addAll(ensureWordAfter);
48  		this.inlineElements.addAll(ensureSentenceBefore);
49  		this.inlineElements.addAll(ensureSentenceAfter);
50  		this.inlineElements.add(wordElement);
51  
52  		this.ensureWordBefore = new HashSet<String>(ensureWordBefore);
53  		this.ensureWordAfter = new HashSet<String>(ensureWordAfter);
54  		this.ensureSentenceBefore = new HashSet<String>(ensureSentenceBefore);
55  		this.ensureSentenceAfter = new HashSet<String>(ensureSentenceAfter);
56  
57  		this.tmpNs = tmpNamespace;
58  	}
59  }
