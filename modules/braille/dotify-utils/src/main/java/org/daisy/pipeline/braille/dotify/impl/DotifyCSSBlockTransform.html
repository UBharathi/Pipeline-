39  public interface DotifyCSSBlockTransform {
40  	
41  	@Component(
42  		name = "org.daisy.pipeline.braille.dotify.impl.DotifyCSSBlockTransform.Provider",
43  		service = {
44  			BrailleTranslatorProvider.class,
45  			TransformProvider.class
46  		}
47  	)
48  	public class Provider extends AbstractTransformProvider<BrailleTranslator> implements BrailleTranslatorProvider<BrailleTranslator> {
49  		
50  		private URI href;
51  		
52  		@Activate
53  		protected void activate(final Map<?,?> properties) {
54  			href = asURI(URLs.getResourceFromJAR("xml/transform/dotify-block-translate.xpl", DotifyCSSBlockTransform.class));
55  		}
56  		
57  		private final static Iterable<BrailleTranslator> empty = Iterables.<BrailleTranslator>empty();
58  		
59  		protected Iterable<BrailleTranslator> _get(Query query) {
60  			final MutableQuery q = mutableQuery(query);
61  			for (Feature f : q.removeAll("input"))
62  				if ("html".equals(f.getValue().get())) {}
63  				else if (!"css".equals(f.getValue().get()))
64  					return empty;
65  			boolean braille = false;
66  			final boolean htmlOut; {
67  				boolean html = false;
68  				for (Feature f : q.removeAll("output"))
69  					if ("css".equals(f.getValue().get())) {}
70  					else if ("html".equals(f.getValue().get()))
71  						html = true;
72  					else if ("braille".equals(f.getValue().get()))
73  						braille = true;
74  					else
75  						return empty;
76  				htmlOut = html;
77  			}
78  			final String locale = q.containsKey("locale") ? q.getOnly("locale").getValue().get() : null;
79  			if (q.containsKey("translator"))
80  				if (!"dotify".equals(q.removeOnly("translator").getValue().get()))
81  					return empty;
82  			q.add("input", "text-css");
83  			if (braille)
84  				q.add("output", "braille");
85  			Iterable<DotifyTranslator> translators = logSelect(q, dotifyTranslatorProvider);
86  			return transform(
87  				translators,
88  				new Function<DotifyTranslator,BrailleTranslator>() {
89  					public BrailleTranslator _apply(DotifyTranslator translator) {
90  						return __apply(
91  							logCreate(new TransformImpl(q.toString(), translator, htmlOut, locale))
92  						);
93  					}
94  				}
95  			);
96  		}
97  		
98  		private class TransformImpl extends AbstractBrailleTranslator {
99  			
100 			private final DotifyTranslator translator;
101 			private final XProc xproc;
102 			
103 			private TransformImpl(String translatorQuery, DotifyTranslator translator, boolean htmlOut, String mainLocale) {
104 				Map<String,String> options = ImmutableMap.of("text-transform", translatorQuery,
105 				                                             // This will omit the <_ style="text-transform:none">
106 				                                             // wrapper. It is assumed that if (output:html) is set, the
107 				                                             // result is known to be braille (which is the case if
108 				                                             // (output:braille) is also set).
109 				                                             "no-wrap", String.valueOf(htmlOut),
110 				                                             "main-locale", mainLocale != null ? mainLocale : "");
111 				xproc = new XProc(href, null, options);
112 				this.translator = translator;
113 			}
114 			
115 			@Override
116 			public XProc asXProc() {
117 				return xproc;
118 			}
119 			
120 			@Override
121 			public ToStringHelper toStringHelper() {
122 				return MoreObjects.toStringHelper("o.d.p.b.liblouis.impl.LiblouisCSSBlockTransform$TransformImpl")
123 					.add("translator", translator);
124 			}
125 		}
126 		
127 		@Reference(
128 			name = "DotifyTranslatorProvider",
129 			unbind = "unbindDotifyTranslatorProvider",
130 			service = DotifyTranslator.Provider.class,
131 			cardinality = ReferenceCardinality.MULTIPLE,
132 			policy = ReferencePolicy.DYNAMIC
133 		)
134 		protected void bindDotifyTranslatorProvider(DotifyTranslator.Provider provider) {
135 			dotifyTranslatorProviders.add(provider);
136 		}
137 		
138 		protected void unbindDotifyTranslatorProvider(DotifyTranslator.Provider provider) {
139 			dotifyTranslatorProviders.remove(provider);
140 			dotifyTranslatorProvider.invalidateCache();
141 		}
142 		
143 		private List<TransformProvider<DotifyTranslator>> dotifyTranslatorProviders
144 		= new ArrayList<TransformProvider<DotifyTranslator>>();
145 		
146 		private TransformProvider.util.MemoizingProvider<DotifyTranslator> dotifyTranslatorProvider
147 		= memoize(dispatch(dotifyTranslatorProviders));
148 		
149 		@Override
150 		public ToStringHelper toStringHelper() {
151 			return MoreObjects.toStringHelper(DotifyCSSBlockTransform.Provider.class.getName());
152 		}
153 	}
