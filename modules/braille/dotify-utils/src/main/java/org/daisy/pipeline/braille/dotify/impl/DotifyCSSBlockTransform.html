39  public interface DotifyCSSBlockTransform {
40  	
41  	@Component(
42  		name = "org.daisy.pipeline.braille.dotify.impl.DotifyCSSBlockTransform.Provider",
43  		service = {
44  			BrailleTranslatorProvider.class,
45  			TransformProvider.class
46  		}
47  	)
48  	public class Provider extends AbstractTransformProvider<BrailleTranslator> implements BrailleTranslatorProvider<BrailleTranslator> {
49  		
50  		private URI href;
51  		
52  		@Activate
53  		private void activate(ComponentContext context, final Map<?,?> properties) {
54  			href = asURI(context.getBundleContext().getBundle().getEntry("xml/transform/dotify-block-translate.xpl"));
55  		}
56  		
57  		private final static Iterable<BrailleTranslator> empty = Iterables.<BrailleTranslator>empty();
58  		
59  		private final static List<String> supportedInput = ImmutableList.of("css");
60  		private final static List<String> supportedOutput = ImmutableList.of("css");
61  		
62  		protected Iterable<BrailleTranslator> _get(Query query) {
63  			final MutableQuery q = mutableQuery(query);
64  			for (Feature f : q.removeAll("input"))
65  				if (!supportedInput.contains(f.getValue().get()))
66  					return empty;
67  			for (Feature f : q.removeAll("output"))
68  				if (!supportedOutput.contains(f.getValue().get()))
69  					return empty;
70  			if (q.containsKey("translator"))
71  				if (!"dotify".equals(q.removeOnly("translator").getValue().get()))
72  					return empty;
73  			q.add("input", "text-css");
74  			Iterable<DotifyTranslator> translators = logSelect(q, dotifyTranslatorProvider);
75  			return transform(
76  				translators,
77  				new Function<DotifyTranslator,BrailleTranslator>() {
78  					public BrailleTranslator _apply(DotifyTranslator translator) {
79  						return __apply(
80  							logCreate(new TransformImpl(q.toString(), translator))
81  						);
82  					}
83  				}
84  			);
85  		}
86  		
87  		private class TransformImpl extends AbstractBrailleTranslator {
88  			
89  			private final DotifyTranslator translator;
90  			private final XProc xproc;
91  			
92  			private TransformImpl(String translatorQuery, DotifyTranslator translator) {
93  				Map<String,String> options = ImmutableMap.of("query", translatorQuery);
94  				xproc = new XProc(href, null, options);
95  				this.translator = translator;
96  			}
97  			
98  			@Override
99  			public XProc asXProc() {
100 				return xproc;
101 			}
102 			
103 			@Override
104 			public ToStringHelper toStringHelper() {
105 				return MoreObjects.toStringHelper("o.d.p.b.liblouis.impl.LiblouisCSSBlockTransform$TransformImpl")
106 					.add("translator", translator);
107 			}
108 		}
109 		
110 		@Reference(
111 			name = "DotifyTranslatorProvider",
112 			unbind = "unbindDotifyTranslatorProvider",
113 			service = DotifyTranslator.Provider.class,
114 			cardinality = ReferenceCardinality.MULTIPLE,
115 			policy = ReferencePolicy.DYNAMIC
116 		)
117 		protected void bindDotifyTranslatorProvider(DotifyTranslator.Provider provider) {
118 			dotifyTranslatorProviders.add(provider);
119 		}
120 		
121 		protected void unbindDotifyTranslatorProvider(DotifyTranslator.Provider provider) {
122 			dotifyTranslatorProviders.remove(provider);
123 			dotifyTranslatorProvider.invalidateCache();
124 		}
125 		
126 		private List<TransformProvider<DotifyTranslator>> dotifyTranslatorProviders
127 		= new ArrayList<TransformProvider<DotifyTranslator>>();
128 		
129 		private TransformProvider.util.MemoizingProvider<DotifyTranslator> dotifyTranslatorProvider
130 		= memoize(dispatch(dotifyTranslatorProviders));
131 	
132 	}
133 }
