98  	@Override
99  	public void run() throws SaxonApiException {
100 		super.run();
101 		try {
102 			
103 			// Read OBFL
104 			ByteArrayOutputStream s = new ByteArrayOutputStream();
105 			Serializer serializer = runtime.getProcessor().newSerializer();
106 			serializer.setOutputStream(s);
107 			serializer.setCloseOnCompletion(true);
108 			serializer.serializeNode(source.read());
109 			serializer.close();
110 			InputStream obflStream = new ByteArrayInputStream(s.toByteArray());
111 			s.close();
112 			
113 			// Convert
114 			// FIXME: duplication with DotifyTaskSystem! => use that class in
115 			// here (like in XMLToOBFL) when it supports setting of the mode
116 			String locale = getOption(_locale).getString();
117 			String mode = getOption(_mode).getString();
118 			String identifier = getOption(_identifier, "");
119 			boolean markCapitalLetters; {
120 				String p = params.get("mark-capital-letters");
121 				markCapitalLetters = (p == null) ? true : !p.equalsIgnoreCase("false"); }
122 			boolean hyphenate; {
123 				String p = params.get("hyphenate");
124 				hyphenate = (p == null) ? true : !p.equalsIgnoreCase("false"); }
125 			boolean allowTextOverflowTrimming; {
126 				String p = params.get("allow-text-overflow-trimming");
127 				allowTextOverflowTrimming = (p == null) ? false : p.equalsIgnoreCase("true"); }
128 			boolean removeStyles; {
129 				String p = params.get("remove-styles");
130 				removeStyles = (p == null) ? false : p.equalsIgnoreCase("true"); }
131 			boolean allowEndingVolumeOnHyphen; {
132 				String p = params.get("allow-ending-volume-on-hyphen");
133 				allowEndingVolumeOnHyphen = (p == null) ? true : p.equalsIgnoreCase("true"); }
134 			FormatterConfiguration.Builder config = FormatterConfiguration.with(locale, mode)
135 				.markCapitalLetters(markCapitalLetters)
136 				.hyphenate(hyphenate)
137 				.allowsTextOverflowTrimming(allowTextOverflowTrimming)
138 				.allowsEndingVolumeOnHyphen(allowEndingVolumeOnHyphen);
139 			if (removeStyles)
140 				config.ignoreStyle("em").ignoreStyle("strong");
141 			
142 			FormatterEngine engine = newFormatterEngine(config.build(), identifier);
143 			s = new ByteArrayOutputStream();
144 			engine.convert(obflStream, s);
145 			obflStream.close();
146 			InputStream pefStream = new ByteArrayInputStream(s.toByteArray());
147 			s.close();
148 			
149 			// Write PEF
150 			result.write(runtime.getProcessor().newDocumentBuilder().build(new StreamSource(pefStream)));
151 			pefStream.close(); }
152 			
153 		catch (Throwable e) {
154 			logger.error("dotify:obfl-to-pef failed", e);
155 			throw new XProcException(step.getNode(), e); }
156 	}

158 	private PagedMediaWriter newPagedMediaWriter(String target) throws PagedMediaWriterConfigurationException {
159 		target = target.toLowerCase();
160 		for (PagedMediaWriterFactoryService s : writerFactoryServices)
161 			if (s.supportsMediaType(target))
162 				return s.newFactory(target).newPagedMediaWriter();
163 		throw new RuntimeException("Cannot find a PagedMediaWriter factory for " + target);
164 	}
165 	
166 	private PagedMediaWriter newPEFWriter(String identifier) throws PagedMediaWriterConfigurationException {
167 		PagedMediaWriter ret = newPagedMediaWriter(MediaTypes.PEF_MEDIA_TYPE);
168 		if (!"".equals(identifier)) {
169 			List<MetaDataItem> meta = new ArrayList<MetaDataItem>();
170 			javax.xml.namespace.QName name = new javax.xml.namespace.QName("http://purl.org/dc/elements/1.1/", "identifier", "dc");
171 			meta.add(new MetaDataItem(name, identifier));
172 			ret.prepare(meta);
173 		}
174 		return ret;
175 	}

177 	private FormatterEngine newFormatterEngine(FormatterConfiguration config, PagedMediaWriter writer) {
178 		return engineFactoryServices.iterator().next().newFormatterEngine(config, writer);
179 	}
180 	
181 	private FormatterEngine newFormatterEngine(FormatterConfiguration config, String identifier) throws PagedMediaWriterConfigurationException {
182 		return newFormatterEngine(config, newPEFWriter(identifier));
183 	}
