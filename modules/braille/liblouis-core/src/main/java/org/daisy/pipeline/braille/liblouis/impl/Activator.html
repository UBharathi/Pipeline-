1   package org.daisy.pipeline.braille.liblouis.impl;
2   
3   import java.net.URI;
4   import java.net.URL;
5   import java.util.Hashtable;
6   
7   import org.daisy.pipeline.braille.common.NativePath;
8   import static org.daisy.pipeline.braille.common.util.URIs.asURI;
9   
10  import org.osgi.framework.BundleActivator;
11  import org.osgi.framework.BundleContext;
12  
13  public class Activator implements BundleActivator {
14  	
15  	private final static boolean LIBLOUIS_EXTERNAL = Boolean.getBoolean("org.daisy.pipeline.braille.liblouis.external");
16  	
17  	public void start(BundleContext context) throws Exception {
18  		if (LIBLOUIS_EXTERNAL) {
19  			NativePath dummy = new NativePath() {
20  				public URI getIdentifier() { return asURI("http://www.liblouis.org/native/EXTERNAL"); }
21  				public URI canonicalize(URI resource) { return null; }
22  				public URL resolve(URI resource) { return null; }
23  				public Iterable<URI> get(String query) { return null; }
24  			};
25  			Hashtable<String,Object> properties = new Hashtable<String,Object>();
26  			properties.put("identifier", "http://www.liblouis.org/native/EXTERNAL");
27  			context.registerService(NativePath.class.getName(), dummy, properties);
28  		}
29  	}
30  	
31  	public void stop(BundleContext context) throws Exception {}
32  	
33  }
