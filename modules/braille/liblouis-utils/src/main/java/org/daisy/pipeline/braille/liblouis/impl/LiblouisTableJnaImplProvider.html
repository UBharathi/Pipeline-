98  	private void registerTableResolver() {
99  		Louis.setTableResolver(new TableResolver() {
100 				private final Map<String,URL> aggregatorTables = new HashMap<String,URL>();
101 				@Override
102 				public URL resolve(String table, URL base) {
103 					logger.debug("Resolving " + table + (base != null ? " against base " + base : ""));
104 					// if we are resolving an include rule from a generated aggregator table, resolve without base
105 					if (aggregatorTables.containsValue(base))
106 						base = null;
107 					File baseFile = base == null ? null : asFile(base); // base is expected to be a file
108 					File[] resolved = tableRegistry.resolveLiblouisTable(new LiblouisTable(table), baseFile);
109 					if (resolved != null) {
110 						logger.debug("Resolved to " + join(resolved, ","));
111 						if (resolved.length == 1)
112 							return asURL(resolved[0]);
113 						else {
114 							// if it is a comma separated table list, create a single file that includes all the sub-tables
115 							if (aggregatorTables.containsKey(table)) {
116 								URL u = aggregatorTables.get(table);
117 								logger.debug("... aggregated into " + u);
118 								return u;
119 							}
120 							try {
121 								StringBuilder b = new StringBuilder();
122 								for (File f : resolved)
123 									b.append("include ").append(asURI(f.getCanonicalFile()).toASCIIString()).append('\n');
124 								InputStream in = new ByteArrayInputStream(b.toString().getBytes(StandardCharsets.UTF_8));
125 								File f = createTempFile("aggregator-", ".tbl");
126 								f.delete();
127 								Files.copy(in, f.toPath());
128 								f = f.getCanonicalFile();
129 								URL u = asURL(f);
130 								aggregatorTables.put(table, u);
131 								logger.debug("... aggregated into " + u);
132 								return u;
133 							} catch (IOException e) {
134 								throw new RuntimeException(e); // should not happen
135 							}
136 						}
137 					}
138 					logger.error("Table could not be resolved");
139 					return null;
140 				}
141 				@Override
142 				public Set<String> list() {
143 					return newHashSet(
144 						transform(
145 							tableRegistry.listAllTableFiles(),
146 							toStringFunction()));
147 				}
148 			}
149 		);

153 	// setLibraryPath, setTableResolver and setLogger are global functions
154 	@Activate
155 	protected void activate() {
156 		logger.debug("Loading liblouis service");
157 		try {
158 			tableRegistry.onPathChange(
159 				new Function<LiblouisTableRegistry,Void>() {
160 					public Void apply(LiblouisTableRegistry r) {
161 						// re-register table resolver so that liblouis-java re-indexes tables
162 						registerTableResolver();
163 						invalidateCache();
164 						return null; }});
165 			registerTableResolver();
166 			// invoke after table resolver registered because otherwise default tables will be unpacked for no reason
167 			logger.debug("liblouis version: {}", Louis.getVersion());
168 			unicodeDisFile = new File(makeUnpackDir(), "unicode.dis");
169 			unpack(
170 				URLs.getResourceFromJAR("/tables/unicode.dis", LiblouisTableJnaImplProvider.class),
171 				unicodeDisFile);
172 			spacesFile = new File(makeUnpackDir(), "spaces.cti");
173 			unpack(
174 				URLs.getResourceFromJAR("/tables/spaces.cti", LiblouisTableJnaImplProvider.class),
175 				spacesFile);
176 			Louis.setLogger(new org.liblouis.Logger() {
177 					@Override
178 					public void log(Level level, String message) {
179 						switch (level) {
180 						case ALL: logger.trace(message); break;
181 						case DEBUG: logger.debug(message); break;
182 						case INFO: logger.debug("INFO: " + message); break;
183 						case WARN: logger.debug("WARN: " + message); break;
184 						// FIXME: capture these and include them into CompilationException or TranslationException
185 						case ERROR: logger.error(message); break;
186 						case FATAL: logger.error(message); break; }}}); }
187 		catch (Throwable e) {
188 			logger.error("liblouis service could not be loaded", e);
189 			throw e; }
190 	}

250 	private TransformProvider<LiblouisTableJnaImpl> _provider
251 	= varyLocale(
252 		new AbstractTransformProvider<LiblouisTableJnaImpl>() {
253 			public Iterable<LiblouisTableJnaImpl> _get(final Query query) {
254 				return Iterables.of(
255 					new WithSideEffect<LiblouisTableJnaImpl,Logger>() {
256 						public LiblouisTableJnaImpl _apply() {
257 							MutableQuery q = mutableQuery(query);
258 							String table = null;
259 							boolean unicode = false;
260 							boolean whiteSpace = false;
261 							boolean display = false;
262 							if (q.containsKey("unicode")) {
263 								q.removeOnly("unicode");
264 								unicode = true; }
265 							if (q.containsKey("white-space")) {
266 								q.removeOnly("white-space");
267 								whiteSpace = true; }
268 							if (q.containsKey("display")) {
269 								q.removeOnly("display");
270 								if (unicode) {
271 									logger.warn("A query with '(unicode)(display)' never matches anything");
272 									throw new NoSuchElementException(); }
273 								display = true; }
274 							if (q.containsKey("table"))
275 								// FIXME: display and remaining features in query are ignored
276 								table = q.removeOnly("table").getValue().get();
277 							else if (q.containsKey("liblouis-table"))
278 								// FIXME: display and remaining features in query are ignored
279 								table = q.removeOnly("liblouis-table").getValue().get();
280 							else if (!q.isEmpty()) {
281 								StringBuilder b = new StringBuilder();
282 								if (display)
283 									b.append("type:display ");
284 								
285 								// FIXME: if !display, need to match for absence of "type:display"
286 								// -> i.e. Liblouis query syntax must support negation!
287 								// -> this used to be solved by matching "type:translation" but the downside of this
288 								//    is that this feature had to be added to every table which is not desired
289 								// -> another solution would be to let Liblouis return a list of possible matches and
290 								//    select the first match that does not end with ".dis" (or that does not have the
291 								//    feature "type:display")
292 								
293 								for (Feature f : q) {
294 									String k = f.getKey();
295 									if (!k.matches("[a-zA-Z0-9_-]+")) {
296 										__apply(
297 											warn("Invalid syntax for feature key: " + k));
298 										throw new NoSuchElementException(); }
299 									b.append(k);
300 									if (f.hasValue()) {
301 										String v = f.getValue().get();
302 										if (!v.matches("[a-zA-Z0-9_-]+")) {
303 											__apply(
304 												warn("Invalid syntax for feature value: " + v));
305 											throw new NoSuchElementException(); }
306 										b.append(":" + v); }
307 									b.append(" "); }
308 								try {
309 									table = Translator.find(b.toString()).getTable(); }
310 								catch (CompilationException e) {}}
311 							if (table != null) {
312 								if (whiteSpace)
313 									table = asURI(spacesFile) + "," + table;
314 								if (unicode)
315 									table = asURI(unicodeDisFile) + "," + table;
316 								try {
317 									return new LiblouisTableJnaImpl(table); }
318 								catch (CompilationException e) {
319 									__apply(
320 										warn("Could not compile table " + table));
321 									logger.warn("Could not compile table", e); }}
322 							throw new NoSuchElementException();
323 						}
324 					}
325 				);
326 			}
327 		}
328 	);
