1   package org.daisy.pipeline.braille.common.calabash.impl;
2   
3   import javax.xml.namespace.QName;
4   import javax.xml.stream.XMLStreamException;
5   import javax.xml.stream.XMLStreamWriter;
6   
7   import static com.xmlcalabash.core.XProcConstants.NS_XPROC_STEP;
8   import com.xmlcalabash.core.XProcException;
9   import com.xmlcalabash.core.XProcRuntime;
10  import com.xmlcalabash.io.WritablePipe;
11  import com.xmlcalabash.library.DefaultStep;
12  import com.xmlcalabash.runtime.XAtomicStep;
13  
14  import net.sf.saxon.s9api.SaxonApiException;
15  
16  import static org.daisy.common.stax.XMLStreamWriterHelper.writeAttribute;
17  import static org.daisy.common.stax.XMLStreamWriterHelper.writeStartElement;
18  import org.daisy.common.xproc.calabash.XProcStep;
19  import org.daisy.common.xproc.calabash.XProcStepProvider;
20  import org.daisy.common.xproc.calabash.XMLCalabashOptionValue;
21  import org.daisy.common.xproc.calabash.XMLCalabashOutputValue;
22  import org.daisy.pipeline.braille.common.Query;
23  import static org.daisy.pipeline.braille.common.Query.util.query;
24  
25  import org.osgi.service.component.annotations.Component;
26  
27  public class PxParseQueryStep extends DefaultStep implements XProcStep {
28  
29  	@Component(
30  		name = "px:parse-query",
31  		service = { XProcStepProvider.class },
32  		property = { "type:String={http://www.daisy.org/ns/pipeline/xproc}parse-query" }
33  	)
34  	public static class StepProvider implements XProcStepProvider {
35  		@Override
36  		public XProcStep newStep(XProcRuntime runtime, XAtomicStep step) {
37  			return new PxParseQueryStep(runtime, step);
38  		}
39  	}
40  
41  	private WritablePipe result = null;
42  	private static final net.sf.saxon.s9api.QName _QUERY = new net.sf.saxon.s9api.QName("query");
43  
44  	private PxParseQueryStep(XProcRuntime runtime, XAtomicStep step) {
45  		super(runtime, step);
46  	}
47  
48  	@Override
49  	public void setOutput(String port, WritablePipe pipe) {
50  		result = pipe;
51  	}
52  
53  	@Override
54  	public void reset() {
55  		result.resetWriter();
56  	}
57  
58  	@Override
59  	public void run() throws SaxonApiException {
60  		try {
61  			serializeQuery(query(new XMLCalabashOptionValue(getOption(_QUERY)).toString()),
62  			               new XMLCalabashOutputValue(result, runtime).asXMLStreamWriter());
63  		} catch (Exception e) {
64  			logger.error("px:parse-query failed", e);
65  			throw new XProcException(step.getNode(), e);
66  		}
67  		super.run();
68  	}
69  
70  	private static final QName C_PARAM_SET = new QName(NS_XPROC_STEP, "param-set", "c");
71  	private static final QName C_PARAM = new QName(NS_XPROC_STEP, "param", "c");
72  	private static final QName _NAME = new QName("name");
73  	private static final QName _NAMESPACE = new QName("namespace");
74  	private static final QName _VALUE = new QName("value");
75  
76  	private static void serializeQuery(Query query, XMLStreamWriter w) throws XMLStreamException {
77  		writeStartElement(w, C_PARAM_SET);
78  		for (Query.Feature f : query) {
79  			writeStartElement(w, C_PARAM);
80  			writeAttribute(w, _NAME, f.getKey());
81  			writeAttribute(w, _NAMESPACE, "");
82  			writeAttribute(w, _VALUE, f.getValue().orElse("true"));
83  			w.writeEndElement();
84  		}
85  		w.writeEndElement();
86  	}
87  }
