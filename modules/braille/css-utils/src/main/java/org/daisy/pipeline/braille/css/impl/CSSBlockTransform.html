90  		private final static Iterable<BrailleTranslator> empty = Iterables.<BrailleTranslator>empty();
91  		
92  		protected Iterable<BrailleTranslator> _get(Query query) {
93  			final MutableQuery q = mutableQuery(query);
94  			for (Feature f : q.removeAll("input"))
95  				if ("html".equals(f.getValue().get())) {}
96  				else if (!"css".equals(f.getValue().get()))
97  					return empty;
98  			boolean braille = false;
99  			final boolean htmlOut; {
100 				boolean html = false;
101 				for (Feature f : q.removeAll("output"))
102 					if ("css".equals(f.getValue().get())) {}
103 					else if ("html".equals(f.getValue().get()))
104 						html = true;
105 					else if ("braille".equals(f.getValue().get()))
106 						braille = true;
107 					else
108 						return empty;
109 				htmlOut = html;
110 			}
111 			final String locale = q.containsKey("locale") ? q.getOnly("locale").getValue().get() : null;
112 			q.add("input", "text-css");
113 			if (braille)
114 				q.add("output", "braille");
115 			Iterable<BrailleTranslator> translators = logSelect(q, translatorProvider);
116 			return transform(
117 				translators,
118 				new Function<BrailleTranslator,BrailleTranslator>() {
119 					public BrailleTranslator _apply(BrailleTranslator translator) {
120 						return __apply(
121 							logCreate(new TransformImpl(translator, false, htmlOut, locale, q))
122 						);
123 					}
124 				}
125 			);
126 		}

133 		private class TransformImpl extends AbstractBrailleTranslator implements XProcStepProvider {
134 			
135 			private final Query mainQuery;
136 			private final BrailleTranslator mainTranslator;
137 			private final boolean forceMainTranslator;
138 			private final Map<String,String> options;
139 			
140 			/**
141 			 * @param mainTranslator translator to be used for the parts of the document that do
142 			 *                       not have their own sub-translator
143 			 * @param forceMainTranslator if true, the main translator is used even if a default
144 			 *                            translator has been defined in CSS
145 			 */
146 			// FIXME: mainTranslator is optional if default translator has been defined in CSS (which we can not know in advance)
147 			private TransformImpl(BrailleTranslator mainTranslator, boolean forceMainTranslator,
148 			                      boolean htmlOut, String mainLocale, Query query) {
149 				options = ImmutableMap.of(// This will omit the <_ style="text-transform:none">
150 				                          // wrapper. It is assumed that if (output:html) is set, the
151 				                          // result is known to be braille (which is the case if
152 				                          // (output:braille) is also set).
153 				                          "no-wrap", String.valueOf(htmlOut),
154 				                          "main-locale", mainLocale != null ? mainLocale : "");
155 				this.mainTranslator = mainTranslator;
156 				this.forceMainTranslator = forceMainTranslator;
157 				mainQuery = query;
158 			}
159 			
160 			@Override
161 			public XProcStep newStep(XProcRuntime runtime, XAtomicStep step) {
162 				return XProcStep.of(
163 					new SingleInSingleOutXMLTransformer() {
164 						public Runnable transform(XMLInputValue<?> source, XMLOutputValue<?> result, InputValue<?> params) {
165 							return () -> {
166 								if (!(source instanceof SaxonInputValue))
167 									throw new IllegalArgumentException();
168 								Mult<SaxonInputValue> mult = ((SaxonInputValue)source).mult(2);
169 								// analyze the input
170 								Map<String,Query> subTranslators
171 									= readTextTransformRules(
172 										mult.get().ensureSingleItem().asNodeIterator().next(),
173 										mainQuery);
174 								BrailleTranslator compoundTranslator;
175 								Function0<Void> evictTempTranslator; {
176 									if (subTranslators != null) {
177 										BrailleTranslator defaultTranslator = mainTranslator;
178 										Query defaultQuery = subTranslators.remove("auto");
179 										if (!forceMainTranslator && defaultQuery != null && !defaultQuery.equals(mainQuery))
180 											defaultTranslator = translatorProvider.get(defaultQuery).iterator().next();
181 										compoundTranslator = new CompoundTranslator(
182 											defaultTranslator,
183 											Maps.transformValues(
184 												subTranslators,
185 												q -> () -> translatorProvider.get(q).iterator().next()));
186 										evictTempTranslator = Provider.this.provideTemporarily(compoundTranslator);
187 									} else {
188 										compoundTranslator = mainTranslator;
189 										evictTempTranslator = Functions.noOp;
190 									}
191 								}
192 								// run the transformation
193 								new CxEvalBasedTransformer(
194 									href,
195 									null,
196 									ImmutableMap.<String,String>builder()
197 									            .putAll(options)
198 									            .put("text-transform",
199 									                 mutableQuery().add("id", compoundTranslator.getIdentifier()).toString())
200 									            .build()
201 								).newStep(runtime, step).transform(
202 									ImmutableMap.of(
203 										new QName("source"), mult.get(),
204 										new QName("parameters"), params),
205 									ImmutableMap.of(
206 										new QName("result"), result)
207 								).run();
208 								evictTempTranslator.apply();
209 							};
210 						}
211 					},
212 					runtime
213 				);
214 			}

223 		private Map<String,Query> readTextTransformRules(Node doc, Query mainQuery) {
224 			if (!(doc instanceof Document))
225 				throw new TransformerException(new IllegalArgumentException());
226 			Map<String,Query> queries = null;
227 			String style = (((Document)doc).getDocumentElement()).getAttribute("style");
228 			if (style != null && !"".equals(style))
229 				for (RuleTextTransform rule : filter(new InlineStyle(style), RuleTextTransform.class))
230 					for (Declaration d : rule)
231 						if (d.getProperty().equals("system")
232 						    && d.size() == 1
233 						    && d.get(0) instanceof TermIdent
234 						    && "braille-translator".equals(((TermIdent)d.get(0)).getValue())) {
235 							MutableQuery query = mutableQuery(mainQuery);
236 							for (Declaration dd : rule)
237 								if (dd.getProperty().equals("system")
238 								    && dd.size() == 1
239 								    && dd.get(0) instanceof TermIdent
240 								    && "braille-translator".equals(((TermIdent)dd.get(0)).getValue()))
241 									;
242 								else if (!dd.getProperty().equals("system")
243 								         && dd.size() == 1
244 								         && (dd.get(0) instanceof TermIdent || dd.get(0) instanceof TermURI)) {
245 									String key = dd.getProperty();
246 									String value;
247 									if (dd.get(0) instanceof TermIdent) {
248 										value = ((TermIdent)dd.get(0)).getValue();
249 										if (query.containsKey(key))
250 											query.removeAll(key);
251 									} else {
252 										URI base = asURI(((TermURI)dd.get(0)).getBase());
253 										if (base == null)
254 											base = asURI(((Document)doc).getBaseURI());
255 										value = base.resolve(((TermURI)dd.get(0)).getValue()).toASCIIString();
256 									}
257 									if (key.equals("contraction") && value.equals("no"))
258 										query.removeAll("grade");
259 									// FIXME: support this in Liblouis translator
260 									if (key.equals("table") || key.equals("liblouis-table")) {
261 										query.removeAll("locale");
262 										query.removeAll("type");
263 										query.removeAll("contraction");
264 										query.removeAll("grade");
265 										query.removeAll("dots");
266 										query.removeAll("direction");
267 									}
268 									query.add(key, value);
269 								} else {
270 									query = null;
271 									break;
272 								}
273 							if (query != null) {
274 								if (queries == null) queries = new HashMap<>();
275 								String name = rule.getName();
276 								queries.put(name == null ? "auto" : name, query);
277 							}
278 							break;
279 						}
280 			return queries;
281 		}
