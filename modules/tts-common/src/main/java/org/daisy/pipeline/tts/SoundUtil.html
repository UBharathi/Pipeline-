1   package org.daisy.pipeline.tts;
2   
3   import java.io.File;
4   import java.io.IOException;
5   
6   import javax.sound.sampled.AudioInputStream;
7   import javax.sound.sampled.AudioSystem;
8   import javax.sound.sampled.UnsupportedAudioFileException;
9   
10  import org.daisy.pipeline.audio.AudioBuffer;
11  
12  public class SoundUtil {
13  	private static int MinRiffHeaderSize = 44;
14  
15  	static public void readWave(File soundFile, AudioBuffer audioBuffer)
16  	        throws UnsupportedAudioFileException, IOException {
17  		int maxLength = (int) (soundFile.length() - MinRiffHeaderSize);
18  		if (maxLength < 0) {
19  			throw new IOException(soundFile.getAbsolutePath()
20  			        + " is too small to be a WAV file");
21  		}
22  
23  		if (maxLength > (audioBuffer.data.length - audioBuffer.size)) {
24  			realloc(audioBuffer, maxLength);
25  		}
26  
27  		AudioInputStream fi = AudioSystem.getAudioInputStream(soundFile);
28  		int read = 0;
29  		while (audioBuffer.size + read != audioBuffer.data.length && read != -1) {
30  			audioBuffer.size += read;
31  			read = fi.read(audioBuffer.data, audioBuffer.size, audioBuffer.data.length
32  			        - audioBuffer.size);
33  		}
34  		fi.close();
35  	}
36  
37  	static public void realloc(AudioBuffer buffer, int extra) {
38  		byte[] newBuffer = new byte[buffer.size + extra];
39  		System.arraycopy(buffer.data, 0, newBuffer, 0, buffer.size);
40  		buffer.data = newBuffer;
41  	}
42  
43  	static public void cancelFootPrint(Iterable<AudioBuffer> buffers,
44  	        AudioBufferAllocator allocator) {
45  		for (AudioBuffer buff : buffers) {
46  			allocator.releaseBuffer(buff);
47  		}
48  	}
49  }
